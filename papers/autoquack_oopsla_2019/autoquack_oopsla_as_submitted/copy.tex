      if (x == 0)
        return 0;
// $\{\p{graph}(\tx{x},\gamma)/| x\neq 0\}$
// $\{\p{graph}(\tx x,\gamma) /| \exists x_0,l,r.~ \gamma(\tx{x}) = (x_0,l,r)\}$
// $\{\p{graph}(\tx x,\gamma) /| \gamma(\tx{x}) = (x_0,l,r)\}$
